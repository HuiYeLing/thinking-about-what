def data_norm(df,*cols):
    df_n = df.copy()
    for col in cols:
        ma = df_n[col].max()
        mi = df_n[col].min()
        df_n[col + '_n'] = (df_n[col] - mi) / (ma - mi)
    return(df_n)
#

def VarianceThreshold(df, threshold=0.):
    print('>>>\n', df.columns.tolist())
    
    # 1 
    var = np.sum(np.power(np.matrix(df.values)-np.matrix(df.mean()), 2), axis=0)/df.shape[0]
    T = []
    # 2 
    for index, v in enumerate(var.reshape(-1, 1)):
        if v > threshold:
            T.append(index)
    df = df.iloc[:, T]
    return var, df
